password() {
    echo "Let's test the password strength."

    while true; do
        echo "Enter your new password (type 'exit' to quit): "
        read -r new_password

        if [ "$new_password" == "exit" ]; then
            echo "Exiting the password checker."
            exit 0
        fi

        if [[ "$new_password" =~ [A-Za-z] ]]; then
            if [[ "$new_password" =~ [0-9] ]]; then
                echo "Password strength: Strong"
            else
                echo "Password strength: Weak "
            fi
            else
                echo "Password strength: Weak "
            fi    
 done
}

anon() {

echo " Please choose 1. Encrypt a file 2. Decrypt a file: "
  read -r answer
  if [ "$answer" = "1" ]; then
    E1
  elif [ "$answer" = "2" ]; then
    D2
  else
    echo "Invalid option, please pick option 1 or 2."
  fi
 
}

E1() { 
   echo "Enter the path to the file you want to encrypt:"
    read -r file_path

    echo "Enter the recipient's GPG key ID or email:"
    read -r recipient

    gpg --recipient "$recipient" --output "${file_path}.gpg" --encrypt "$file_path"

    echo "File encrypted successfully: ${file_path}.gpg"
}

D2() {

    echo "Enter the path to the encrypted file:"
    read -r encrypted_file_path

    gpg --output "${encrypted_file_path%.gpg}" --decrypt "$encrypted_file_path"

    echo "File decrypted successfully: ${encrypted_file_path%.gpg}"
}
